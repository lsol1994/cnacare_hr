Apps Script 설치 가이드

const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ok: true, message: '씨앤에이케어 연결 성공!'})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const d = JSON.parse(e.postData.contents); const type = d.type; // ── 환자관리 시트 ── if (type === 'patient') { const sh = SHEET_ID.getSheetByName('씨앤에이케어(환자관리)') || SHEET_ID.getSheetByName('환자관리') || SHEET_ID.insertSheet('환자관리'); if (sh.getLastRow() === 0) { sh.appendRow([ 'NO.','날짜','성함','주민등록번호','성별','연락처', '병원(병동,호수)','질환/상태','특이사항/요청사항', '매칭 간병인','진행상태','간병비','예정기간', '시작일','시작시간','마감일','마감시간' ]); } const no = sh.getLastRow(); // 마지막 행 번호를 NO로 사용 sh.appendRow([ no, d.createdAt || new Date().toLocaleString('ko-KR'), d.name || '', '', // 주민번호 (환자는 미수집) d.gender || '', String(d.contact || ''), // 전화번호 텍스트 d.hospital || '', [d.diagnosis, d.surgery !== '수술 없음' ? d.surgery : '', d.diaper ? '기저귀착용' : '', d.move ? '거동가능' : '거동불가' ].filter(Boolean).join(' / '), d.request || d.extra || '', d.caregiverName || '', d.status || '모집중', d.pay ? d.pay + '원' : '', d.duration || '', d.startDate || '', d.startTime || '', '', '' ]); } // ── 간병인관리 시트 ── else if (type === 'caregiver') { const sh = SHEET_ID.getSheetByName('씨앤에이케어(간병인관리)') || SHEET_ID.getSheetByName('간병인관리') || SHEET_ID.insertSheet('간병인관리'); if (sh.getLastRow() === 0) { sh.appendRow([ 'NO.','날짜','근무상태','성함','주민등록번호','나이','성별', '간병비','연락처','은행명','계좌번호','예금주','거주지(시,군,구)', '병원','시작일','시작시간','마감일','마감시간', '경력','휴무희망일','특화분야','사고이력','보호자 피드백' ]); } const no = sh.getLastRow(); sh.appendRow([ no, d.regDate || new Date().toLocaleString('ko-KR'), d.workStatus || '', d.name || '', d.ssn || '', d.age || '', d.gender || '', d.pay ? d.pay + '원' : '', String(d.phone || ''), // 전화번호 텍스트 d.bankName || '', d.bankAccount || '', d.bankHolder || '', d.region || '', d.hospital || '', d.startDate || '', d.startTime || '', d.endDate || '', d.endTime || '', d.career || '', d.dayOff || '', d.specialty || '', d.accident || '', d.feedback || '' ]); } // ── 간병일지 시트 ── else if (type === 'journal') { const sh = SHEET_ID.getSheetByName('간병일지') || SHEET_ID.insertSheet('간병일지'); if (sh.getLastRow() === 0) { sh.appendRow([ '간병인 성명','간병인 연락처','환자병원', '간병일자(시작)','시작시간','간병일자(마감)','마감시간','비고' ]); } sh.appendRow([ d.caregiverName || '', String(d.phone || ''), d.hospital || '', d.startDate || '', d.startTime || '', d.endDate || '', d.endTime || '', d.note || '' ]); } // ── 환자 상태 업데이트 ── else if (type === 'updatePatient') { const sh = SHEET_ID.getSheetByName('씨앤에이케어(환자관리)') || SHEET_ID.getSheetByName('환자관리'); if (!sh) throw new Error('환자관리 시트를 찾을 수 없어요'); const data2 = sh.getDataRange().getValues(); for (let i = 1; i < data2.length; i++) { if (data2[i][2] === d.name && data2[i][13] === d.startDate) { sh.getRange(i+1, 10).setValue(d.caregiverName || ''); // 매칭간병인 sh.getRange(i+1, 11).setValue(d.status || ''); // 진행상태 break; } } } return ContentService .createTextOutput(JSON.stringify({ok: true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok: false, error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
